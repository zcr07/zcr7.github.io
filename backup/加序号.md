## 加序号


```

CapsLock & v::
    if (!bzz)
    bzz := 2                      ; 从2开始
    Send, {Enter} 
    SendInput, {Text}%bzz%.           ; 用{Text}确保输出字面符号
    Send, {Space}                     ; 再单独发送空格
    bzz++
return

;ΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞ   CapsLock & b  1.2.3.到无穷   ΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞ 007-2886

CapsLock & b::
    if (!bz)
        bz := "b"                                    ; 如果变量 bz 尚未初始化，则初始化为 "b"
    Send, {Enter} 
    SendInput, {Text}%bz%.           ; 用{Text}确保输出字面符号
    Send, {Space}                     ; 再单独发送空格

    ; 字母递增逻辑                                 判断是否在 'a-z' 范围 ASCII码a-z 97-122
    if (StrLen(bz) == 1 && Asc(bz) >= 97 && Asc(bz) < 122) 
        bz := Chr(Asc(bz) + 1)  ; 递增到下一个字母
    else if (bz == "z")
        bz := "a"  ; 如果当前是 'z'，循环回到 'a'
Return
;ΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞ   CapsLock & b   a-z 循环   ΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞΞ 007-2886

CapsLock & n::
    	if (!azz)
        	azz := "2"                           ; 如果变量 azz 尚未初始化，则初始化为 "2"
    Send, {Enter} 
    SendInput, {Text}%azz%.           ; 用{Text}确保输出字面符号
    Send, {Space}   

; 更新数字，如果超过9则重置为1
azz := (azz < 9) ? azz + 1 : 1
Return
```
